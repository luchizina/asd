package Logica;

import java.util.List;
import java.util.ArrayList;
import Logica.Persona;
import Persistencia.DBPersona;
/**
 *
 * @author apias
 */
public class ControladorPersona implements IControladorPersona {
    private List<Persona> personas;
    private DBPersona dbPersona=null;

    public ControladorPersona(){
        this.personas=new ArrayList<Persona>();
        this.dbPersona=new DBPersona();
    }
    @Override
    public boolean AgregarPersona(String nombre, int edad) {
        Persona p=new Persona(nombre, edad);
        this.personas.add(p);
        return this.dbPersona.agregarPersona(p);
    }

    @Override
    public boolean BorrarPersona(DataPersona p) {
        //Persona p=new Persona(nombre, edad);
        //this.personas.add(p);
        //return this.dbPersona.agregarPersona(p);
        return this.dbPersona.borrarPersona(new Persona(p.getNombre(),p.getEdad()));
    }
    
    
    @Override
    public List<DataPersona> BuscarPersonas(String nombre) {
        List<DataPersona> retornar=new ArrayList<DataPersona>();
        List<Persona> auxiliar;
        //throw new UnsupportedOperationException("Not supported yet."); //To change body of generated methods, choose Tools | Templates.
        auxiliar=this.dbPersona.buscarPersonas(nombre);
        for(int i=0; i<auxiliar.size(); i++)
            retornar.add(new DataPersona(auxiliar.get(i)));
        return retornar;
    }
}
