package Logica;

/**
 *
 * @author apias
 */
public class DataPersona {
    private String nombre;
    private int edad;
    public DataPersona(){
        this.edad=0;
        this.nombre="";
    }
    public DataPersona(Persona p){
        this.edad=p.getEdad();
        this.nombre=p.getNombre();
    } 
    public DataPersona(String nombre, int edad){
        this.edad=edad;
        this.nombre=nombre;
    }    

    public String getNombre() {
        return nombre;
    }

    public void setNombre(String nombre) {
        this.nombre = nombre;
    }

    public int getEdad() {
        return edad;
    }

    public void setEdad(int edad) {
        this.edad = edad;
    }
    
}
